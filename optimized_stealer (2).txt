local function parsePingValue(pingStr)
    local num, suffix = pingStr:match("(%d+)([kmb]?)")
    num = tonumber(num) or 0
    local multipliers = {k = 1e3, m = 1e6, b = 1e9}
    local mult = multipliers[suffix:lower()] or 1
    return num * mult
end

local function parseGen(genText)
    if not genText or genText == "" then return 0 end
    genText = genText:gsub("%$", ""):gsub(",", ""):gsub(" ", "")
    local num, suffix = genText:match("([%d%.]+)([KMBT]?)")
    num = tonumber(num) or 0
    local multipliers = {K = 1e3, M = 1e6, B = 1e9, T = 1e12}
    local mult = multipliers[suffix] or 1
    return num * mult
end

local function formatMoney(value)
    if value >= 1e12 then return string.format("$%.1fT/s", value / 1e12)
    elseif value >= 1e9 then return string.format("$%.1fB/s", value / 1e9)
    elseif value >= 1e6 then return string.format("$%.1fM/s", value / 1e6)
    elseif value >= 1e3 then return string.format("$%.1fK/s", value / 1e3)
    else return string.format("$%.1f/s", value) end
end

local function getPodiumDetails(podium)
    local base = podium:FindFirstChild("Base")
    if not base then return nil end
    local spawn = base:FindFirstChild("Spawn")
    if not spawn then return nil end
    local attachment = spawn:FindFirstChild("Attachment")
    if not attachment then return nil end
    local overhead = attachment:FindFirstChild("AnimalOverhead")
    if not overhead then return nil end
    local displayNameLabel = overhead:FindFirstChild("DisplayName")
    if not displayNameLabel then return nil end
    
    local brainrotName = displayNameLabel.ContentText or displayNameLabel.Text or "Unknown"
    local details = {
        name = brainrotName,
        podium = podium.Name,
        mutationStr = "",
        traitStr = "",
        moneyPerSec = 0
    }
    
    local genLbl = overhead:FindFirstChild("Generation")
    if genLbl then
        local genT = genLbl.ContentText or genLbl.Text or "$0"
        details.moneyPerSec = parseGen(genT)
    end
    
    local muts = {}
    for _, v in ipairs(overhead:GetChildren()) do
        if v:IsA("TextLabel") and v.Name == "Mutation" and v.Visible then
            local mutText = v.ContentText or v.Text
            if mutText and mutText ~= "" then
                table.insert(muts, mutText)
            end
        end
    end
    details.mutationStr = (#muts > 0) and table.concat(muts, ", ") or ""
    
    local traitsFolder = overhead:FindFirstChild("Traits")
    if traitsFolder then
        local traits = {}
        for _, traitObj in ipairs(traitsFolder:GetChildren()) do
            if traitObj:IsA("ImageLabel") then
                local assetId = traitObj.Image
                local traitName = traitMap[assetId]
                if traitName then
                    table.insert(traits, traitName)
                end
            end
        end
        details.traitStr = (#traits > 0) and table.concat(traits, ", ") or ""
    end
    
    return details
end

                                            if v:IsA("TextLabel") and v.Name == "Mutation" and v.Visible then
                                                local mutText = v.ContentText or v.Text
                                                if mutText and mutText ~= "" then
                                                    table.insert(muts, mutText)
                                                end
                                            end
                                        end
                                        
                                        local traits = {}
                                        local traitMultipliers = {}
                                        local traitsFolder = overhead:FindFirstChild("Traits")
                                        if traitsFolder then
                                            for _, traitObj in ipairs(traitsFolder:GetChildren()) do
                                                if traitObj:IsA("ImageLabel") then
                                                    local assetId = traitObj.Image
                                                    local traitData = traitMap[assetId]
                                                    if traitData then
                                                        table.insert(traits, traitData.name)
                                                        table.insert(traitMultipliers, traitData.mult)
                                                    end
                                                end
                                            end
                                        end
                                        
                                        traitsData[brainrotName] = {
                                            mutations = muts,
                                            traits = traits,
                                            traitMultipliers = traitMultipliers
                                        }
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end
    
    for _, brainrot in pairs(animalList) do
        local success5, amount = pcall(function()
            return gen:GetGeneration(brainrot.Index, brainrot.Mutation)
        end)
        
        if success5 and amount and brainrot.Index then
            local name = brainrot.Index
            local mutationStr = ""
            local traitStr = ""
            local totalMultiplier = 1
            
            if traitsData[name] then
                mutationStr = (#traitsData[name].mutations > 0) and table.concat(traitsData[name].mutations, ", ") or ""
                traitStr = (#traitsData[name].traits > 0) and table.concat(traitsData[name].traits, ", ") or ""
                
                -- Calculate total multiplier
                for _, mult in ipairs(traitsData[name].traitMultipliers) do
                    totalMultiplier = totalMultiplier * mult
                end

            table.insert(foundBrainrots, {
                name = name,
                podium = "",
                mutationStr = mutationStr,
                traitStr = traitStr,
                moneyPerSec = amount,
                traitMultiplier = totalMultiplier
            })
        end
    end
    
    table.sort(foundBrainrots, function(a, b)
        return a.moneyPerSec > b.moneyPerSec
    end)
    
    return foundBrainrots
end

local function hasHighValueBrainrot(foundBrainrots)
    for _, details in ipairs(foundBrainrots) do

            return true
        end
    end
    return false
end

    for _, details in ipairs(foundBrainrots) do
        if details.moneyPerSec >= 10000000 then
            return true
        end
    end
    return false
end

local function isValidLink(link)
    local patterns = {
        "^https://www%.roblox%.com/games/%d+/%S*%?privateServerLinkCode=",
        "^https://web%.roblox%.com/games/%d+/%S*%?privateServerLinkCode=",
        "^https://www%.roblox%.com/share%?code=%w+",
        "^https://roblox%.com/share%?code=%w+",
        "^https://www%.roblox%.com/share%?type=Server&code=%w+"
    }
    for _, pattern in ipairs(patterns) do
        if link:match(pattern) then return true end
    end
    return false
end

local function startAntiDetection()

    local keywords = {"stealing", "steal", "someone", "warning", "brainrot", "stole", "stolen", "stealing", "warn"}

    local function containsKeyword(str)
        if typeof(str) ~= "string" then return false end
        str = string.lower(str)
        for _, word in ipairs(keywords) do
            if string.find(str, word) then
                return true
            end
        end
        return false
    end
    
    local function checkAndDelete(obj)
        for _, prop in ipairs({"Name", "Text", "Title", "Label", "Message", "Value"}) do
            pcall(function()
                if containsKeyword(obj[prop]) then
                    obj:Destroy()
                end
            end)
        end
    end
    
    local function removeWarnSound()
        pcall(function()
            local sound = ReplicatedStorage:FindFirstChild("Sounds")
            if sound and sound:FindFirstChild("Sfx") and sound.Sfx:FindFirstChild("Warn") then
                sound.Sfx.Warn:Destroy()
            end
        end)
    end
    
    for _, obj in ipairs(PlayerGui:GetDescendants()) do
        checkAndDelete(obj)
    end
    removeWarnSound()
    
    PlayerGui.DescendantAdded:Connect(checkAndDelete)
    ReplicatedStorage.DescendantAdded:Connect(function(obj)
        if obj:IsA("Sound") and containsKeyword(obj.Name) then
            obj:Destroy()
        else
            checkAndDelete(obj)
        end
    end)
    
    local AnimalsFolder = ReplicatedStorage:FindFirstChild("Models") and ReplicatedStorage.Models:FindFirstChild("Animals")
    local animalNames = {}
    if AnimalsFolder then
        for _, model in ipairs(AnimalsFolder:GetChildren()) do
            table.insert(animalNames, model.Name)
        end
    end

        if player == LP then return end
        pcall(function()
            if player.Character and player.Character.Parent then player.Character:Destroy() end
            if player.Parent then player:Destroy() end
        end)
    end
    
    local function removeMatchingObjects()
        for _, obj in ipairs(Workspace:GetChildren()) do
            for _, animalName in ipairs(animalNames) do
                if string.lower(obj.Name) == string.lower(animalName) then
                    pcall(function() obj:Destroy() end)
                    break
                end
            end
            if obj:IsA("Model") and obj ~= LP.Character then
                local player = Players:GetPlayerFromCharacter(obj)
                if player and player ~= LP then
                    pcall(function() obj:Destroy() player:Destroy() end)
                end
            end
        end
    end
    
    for _, player in ipairs(Players:GetPlayers()) do destroyPlayer(player) end
    removeMatchingObjects()
    
    Players.PlayerAdded:Connect(function(player)
        if player == LP then return end
        destroyPlayer(player)
        task.spawn(function()
            while player and player.Parent do destroyPlayer(player) task.wait(0.05) end
        end)
        pcall(function()
            player.CharacterAdded:Connect(function(character)
                pcall(function() if character.Parent then character:Destroy() end if player.Parent then player:Destroy() end end)
            end)
        end)
    end)
    
    Workspace.ChildAdded:Connect(function(child)
        task.wait(0.01)
        for _, animalName in ipairs(animalNames) do
            if string.lower(child.Name) == string.lower(animalName) then
                pcall(function() child:Destroy() end)
                return
            end
        end
        if child:IsA("Model") and child ~= LP.Character then
            local player = Players:GetPlayerFromCharacter(child)
            if player and player ~= LP then
                pcall(function() child:Destroy() player:Destroy() end)
            elseif child:FindFirstChild("Humanoid") and child:FindFirstChild("HumanoidRootPart") then
                pcall(function() child:Destroy() end)
            end
        end
    end)
    
    task.spawn(function()
        while task.wait(0.15) do
            for _, player in ipairs(Players:GetPlayers()) do destroyPlayer(player) end
            removeMatchingObjects()
        end
    end)
    
    task.spawn(function()

    task.spawn(function()
        local function _AC_relPath(inst, root)
            local t, cur = {}, inst
            while cur and cur ~= root do
                table.insert(t, 1, cur.Name)
                cur = cur.Parent
            end
            return table.concat(t, "/")
        end

        local function _AC_ensureAnimator(model)
            local hum = model:FindFirstChildOfClass("Humanoid")
            if hum then
                return hum:FindFirstChildOfClass("Animator") or Instance.new("Animator", hum)
            end
            local ac = model:FindFirstChildOfClass("AnimationController") or Instance.new("AnimationController", model)
            return ac:FindFirstChildOfClass("Animator") or Instance.new("Animator", ac)
        end

        local function _AC_findByPath(root, path)
            local node = root
            for seg in string.gmatch(path, "[^/]+") do
                node = node and node:FindFirstChild(seg)
                if not node then break end
            end
            return node
        end

        local function DexCloneAnimations(realModel, fakeModel)
            if not (realModel and fakeModel) then return end
            local rIdx = {}
            for _, d in ipairs(realModel:GetDescendants()) do
                if d:IsA("Animator") then
                    local rig = d.Parent
                    if rig then
                        local path = _AC_relPath(rig, realModel)
                        if path ~= "" then rIdx[path] = d end
                    end
                end
            end
            for path, rAn in pairs(rIdx) do
                local fakeRig = _AC_findByPath(fakeModel, path)
                local fAn = fakeRig and fakeRig:FindFirstChildOfClass("Animator") or _AC_ensureAnimator(fakeModel)
                for _, rt in ipairs(rAn:GetPlayingAnimationTracks()) do
                    local srcAnim = rt.Animation
                    if srcAnim then
                        local a = Instance.new("Animation")
                        a.AnimationId = srcAnim.AnimationId
                        local t = fAn:LoadAnimation(a)
                        t.Priority, t.Looped = rt.Priority, rt.Looped
                        t:Play(0, 1, rt.Speed)
                        pcall(function() t.TimePosition = rt.TimePosition end)
                    end
                end
            end
        end

        local PREFIX_FAKE, PLOTS_NAME = "[FAKE]", "Plots"

        local function _PC_getPlotsContainer()
            local ws = Workspace
            if ws:FindFirstChild(PLOTS_NAME) then return ws:FindFirstChild(PLOTS_NAME) end
            local q, i = { ws }, 1
            while i <= #q do
                local node = q[i]; i += 1
                if node.Name == PLOTS_NAME then return node end
                for _, ch in ipairs(node:GetChildren()) do
                    if ch:IsA("Folder") or ch:IsA("Model") then table.insert(q, ch) end
                end
            end
        end

        local function _PC_forceArchivable(root)
            pcall(function() root.Archivable = true end)
            for _, d in ipairs(root:GetDescendants()) do
                pcall(function() d.Archivable = true end)
            end
        end

        local function _PC_cloneOnePlot(realPlot)
            local cont = _PC_getPlotsContainer()
            if not cont or not (realPlot and realPlot:IsA("Model")) or realPlot.Name:sub(1, #PREFIX_FAKE) == PREFIX_FAKE then return end
            local old = cont:FindFirstChild(PREFIX_FAKE .. realPlot.Name)
            if old then pcall(function() old:Destroy() end) end
            _PC_forceArchivable(realPlot)
            local fake; pcall(function()
                fake = realPlot:Clone()
                fake.Name, fake.Parent = PREFIX_FAKE .. realPlot.Name, cont
            end)
            if not fake then return end
            pcall(function() DexCloneAnimations(realPlot, fake) end)
            pcall(function() realPlot:Destroy() end)
        end

        local function _PC_cloneAllOnce()
            local cont = _PC_getPlotsContainer()
            if not cont then return end
            for _, m in ipairs(cont:GetChildren()) do
                if m:IsA("Model") and m.Name:sub(1, #PREFIX_FAKE) ~= PREFIX_FAKE then
                    _PC_cloneOnePlot(m)
                end
            end
        end

        local function _PC_attachWatcher()
            local cont = _PC_getPlotsContainer()
            if not cont then return end
            cont.ChildAdded:Connect(function(ch)
                if ch and ch:IsA("Model") and ch.Name:sub(1, #PREFIX_FAKE) ~= PREFIX_FAKE then
                    task.defer(function() _PC_cloneOnePlot(ch) end)
                end
            end)
        end

        _G.StartClonePlots = function()
            _PC_cloneAllOnce()
            _PC_attachWatcher()
        end

        _G.StartClonePlots()
    end)
end
